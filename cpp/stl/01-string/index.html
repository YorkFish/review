#include <iostream>
#include <cstdio>

using namespace std;

int main()
{
    string s1;                  // 默认的空字符串
    string s2 = s1;             // s2 是 s1 的一个副本
    string s3 = "york";         // s3 是字符串字面值的一个副本
    string s4(10, 'c');         // 可以看成是 Python 的 'c' * 10

    printf("%s\n", s3.c_str()); // 函数 c_str() 返回的是 s3 的首地址
    printf("%s\n", s4.c_str());

    return 0;
}

>>>

york
cccccccccc

ps

  • GNU 的编译器不导入 <string> 也能使用它，微软的编译器需要导入
  • string 可以用 printf() 输出，但不能用 scanf() 读入

3. 判空与长度

  • empty()
  • size(), length()
    • 两者都是 “Return length of string”
    • 并不是像 strlen() 那样用一次计算一次，而是有一个变量存字符串的大小
    • 两者的值是一样的，可以在理解上作细微区别
      • size() 一般用作返回容器大小的方法
      • length() 一般用作返回一个序列的长度
    • s1 > s2, s1 < s2, s1 == s2, s1 >= s2, s1 <= s2, s1 != s2 六种比较方式可以直接用
    • string 可以和 "." 这种字符串比较，但不可以和 '.' 这种字符比较

using namespace std;

int main()
{
    string s = "york";

    cout << s.size() << endl;
    cout << s.length() << endl;

    return 0;

int main()
{
    string name = "york";

using namespace std;

void print(char c)
{

#include <iostream>

using namespace std;

int main()
{
    string s = "york";
    for (string::iterator it = s.begin(); it != s.end(); it ++ )
        cout << *it << endl;  // * 是解引用

    return 0;
}

>>>

y
o
r
k

ps

  • 可以把迭代器看成指针
  • 可以认为 s.end() 指向 '\0'
  • 这里的 for 中使用 it < s.end() 也行，不过 != 比较保险，下面的不能用 <
    1. <set>, <map>，因为它们的 -- 与 ++ 找的是前驱与后继
    2. <unordered_set>, <unordered_map>，因为它们是无序的
  • 类型名比较长时，可以使用 auto，如 for 中的 it 可以定义成 auto it
  • auto 也是 C++11 增加的，可以把它当作是 golang 的自动推导

9. substr

  • 与 Python 的切片不一样
    • 默认：从开头到结尾
    • 一个参数，表示起点：从起点到结尾
    • 两个参数
      • 起点与步长：从起点开始，共取步长个字符
      • 若第二个参数是负数，不是 Python 那样反着数，均表示“到结尾”

#include <iostream>

using namespace std;

int main()
{
    string s = "yorkfish";
    cout << s.substr() << endl;         // s[:]
    cout << s.substr(4) << endl;        // s[4:]
    cout << s.substr(4, -1) << endl;    // s[4:]
    cout << s.substr(4, -5) << endl;    // s[4:]
    cout << s.substr(2, 4) << endl;     // s[2:2 + 4]

    return 0;
}

#include <iostream>
#include <sstream>
#include <cstdio>

using namespace std;

int main()
{
    char str[1000], name[100];
    int width, height;
    double area;
    fgets(str, 1000, stdin);
    sscanf(str, "%d%s%d%lf", &width, name, &height, &area);
    cout << width << endl
         << height << endl
         << name << endl
         << area << endl;

    return 0;
}

#include <iostream>
#include <sstream>

using namespace std;

int main()
{
    string s, word;
    getline(cin, s);
    stringstream ssin(s);   // 字符流，这一句与下面两句一个意思
    // stringstream ssin;
    // ssin << s;
    
    while (ssin >> word)    // 当 cin 用
    {
        if (word == "york") break;
        else cout << word << ' ';
    }
    cout << endl;

    return 0;
}

  • 输入

    fish1 fish2 fish3 york

  • 输出

    fish1 fish2 fish3
