鱼塘钓鱼
2021/02/09    3 min read    YorkFish
  • 活动时间：2021-02-08

题目描述

  • AcWing 1262
  • 简单
  • 有 N 个鱼塘排成一排，每个鱼塘中有一定数量的鱼，例如：N=5 时，如下表
鱼塘编号 1 2 3 4 5
第 1 分钟能钓到的鱼的数量(1…1000) 10 14 20 16 9
每钓鱼 1 分钟钓鱼数的减少量(1…100) 2 4 6 5 3
当前鱼塘到下一个相邻鱼塘需要的时间（单位：分钟） 3 5 4 4 /
  • 即：在第 1 个鱼塘中钓鱼第 1 分钟内可钓到 10 条鱼，第 2 分钟内只能钓到 8 条鱼，……，第 5 分钟以后再也钓不到鱼了。
  • 从第 1 个鱼塘到第 2 个鱼塘需要 3 分钟，从第 2 个鱼塘到第 3 个鱼塘需要 5 分钟，……
  • 给出一个截止时间 T，设计一个钓鱼方案，从第 1 个鱼塘出发，希望能钓到最多的鱼。
  • 假设能钓到鱼的数量仅和已钓鱼的次数有关，且每次钓鱼的时间都是整数分钟。

输入输出

输入格式

  • 共 5 行，分别表示
    1. 第 1 行为 N；
    2. 第 2 行为第 1 分钟各个鱼塘能钓到的鱼的数量，每个数据之间用一空格隔开；
    3. 第 3 行为每过 1 分钟各个鱼塘钓鱼数的减少量，每个数据之间用一空格隔开；
    4. 第 4 行为当前鱼塘到下一个相邻鱼塘需要的时间；
    5. 第 5 行为截止时间 T

输出格式

  • 一个整数（不超过 231−1），表示你的方案能钓到的最多的鱼。

数据范围

  • 1 <= N <= 100
  • 1 <= T <= 1000

样例

输入样例

5
10 14 20 16 9
2 4 6 5 3
3 5 4 4
14

输出样例

76

默认代码

  • 无

Solution

  • 主要是多路归并，大顶堆可用可不用

C++

  • 即使有 100 个鱼塘，1000 分钟
  • 遍历最终池塘
    • 遍历剩余时间
      • 遍历之前的池塘得到最大值
  • O(n*n*T) = 107 可以接受

#include <iostream>
#include <cstring>

using namespace std;

const int N = 110;

int a[N], d[N], l[N];  // 第一分钟，衰减，换位时间
int spend[N];

int get(int k) {
    return max(0, a[k] - d[k] * spend[k]);
}

int work(int n, int T) {
    int res = 0;
    memset(spend, 0, sizeof spend);
    for (int i = 0; i < T; i ++ ) {
        int k = 1;
        for (int j = 1; j <= n; j ++ )
            if (get(k) < get(j))
                k = j;
        res += get(k);
        spend[k] ++ ;
    }
    return res;
}

int main() {
    int n, T;
    cin >> n;
    for (int i = 1; i <= n; i ++ ) cin >> a[i];
    for (int i = 1; i <= n; i ++ ) cin >> d[i];
    for (int i = 2; i <= n; i ++ ) {
        cin >> l[i];
        l[i] += l[i - 1];
    }
    cin >> T;

    int res = 0;
    for (int i = 1; i <= n; i ++ ) res = max(res, work(i, T - l[i]));
    cout << res << endl;

    return 0;
}

#include <iostream>
#include <queue>

#define x first
#define y second

using namespace std;

typedef pair<int, int> PII;
const int N = 110;

int a[N], d[N], l[N];  // 第一分钟，衰减，换位时间

int main() {
    int n, T;
    cin >> n;
    for (int i = 1; i <= n; i++) cin >> a[i];
    for (int i = 1; i <= n; i++) cin >> d[i];
    for (int i = 2; i <= n; i++) {
        cin >> l[i];
        l[i] += l[i - 1];
    }
    cin >> T;

    int res = 0;
    for (int i = 1; i <= n; i ++ ) {
        int fish_time = T - l[i];
        priority_queue<PII> heap;
        for (int j = 1; j <= i; j ++ ) heap.push({a[j], d[j]});

        int fish = 0;
        while (heap.size() && fish_time) {
            PII t = heap.top();
            heap.pop();
            fish_time -- ;
            fish += t.x;
            t.x -= t.y;
            if (t.x > 0) heap.push({t.x, t.y});
        }
        res = max(res, fish);
    }
    cout << res << endl;

    return 0;
}

def get(k):
    return max(0, first[k] - decay[k] * spend[k])


def work(pools, time):
    for i in range(N):
        spend[i] = 0
    res = 0
    for i in range(time):
        k = 1
        for j in range(1, pools + 1):
            if get(k) < get(j):
                k = j
        res += get(k)
        spend[k] += 1
    return res


if __name__ == "__main__":
    N = 110
    n = int(input())
    first = [0] + list(map(int, input().split()))
    decay = [0] + list(map(int, input().split()))
    walk = [0, 0] + list(map(int, input().split()))
    t = int(input())
    for i in range(3, n + 1):
        walk[i] += walk[i - 1]
    spend = [0] * N

    res = 0
    for i in range(1, n + 1):
        res = max(res, work(i, t - walk[i]))
    print(res)

from heapq import heapify, heappush, heappop

n = int(input())
first = list(map(int, input().split()))
decay = list(map(int, input().split()))
walk = [0] + list(map(int, input().split()))
t = int(input())

res = 0  # 默认“空军”
for k in range(n):
    heap = [[-first[i], decay[i]] for i in range(k + 1)]  # Python 只有“小堆”，所以传负值
    heapify(heap)
    t -= walk[k]
    fish = 0
    for j in range(t):
        f, d = heappop(heap)
        if f >= 0:
            break
        fish -= f
        heappush(heap, [f + d, d])
    res = max(res, fish)
print(res)
